(defn write!
  ([content]
   (write! config/DEFAULT-LOG-FILEPATH content))

  ([filepath content]
   (let [timestamp (time/timestamp-string)
         output    (str timestamp " " content)]
        (io/prepend-to-file! filepath output {:create? true :max-line-count config/DEFAULT-MAX-LINE-COUNT}))))
